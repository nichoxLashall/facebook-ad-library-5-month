# Parsing the directory structure and splitting by lines to process it
lines = directory_structure.strip().split("\n")

# Files and folders that need to be processed
files = []

# Iterate over the lines to capture all the paths
current_path = ""
for line in lines:
 if line.startswith("├──") or line.startswith("└──"):
 # Extract the file name and build the path
 file_name = line.strip("├── └──").strip()
 if current_path:
 file_path = f"{current_path}/{file_name}"
 else:
 file_path = file_name
 files.append(file_path)
 elif line.endswith("/"):
 # It's a directory, set the current path
 current_path = line.strip("├── └──").strip()
 else:
 # Reset path when we hit another root directory
 current_path = ""

print(files)
